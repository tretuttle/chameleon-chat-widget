Amigo Chat Widget - Responsive Iframe Examples

1. Responsive Mode (Recommended)

This mode adapts to the container size and works great for embedding in websites:

HTML Code:

<iframe 
    class="responsive-iframe"
    src="https://tretuttle.github.io/chameleon-chat-widget/widget?mode=responsive&state=horizontal"
    style="width: 100%; min-height: 400px; border: none; border-radius: 8px;"
    title="Amigo Chat Widget">
</iframe>

2. Fixed Size Mode

Use this mode when you want a consistent widget size:

HTML Code:

<iframe 
    src="https://tretuttle.github.io/chameleon-chat-widget/widget?mode=fixed&state=horizontal"
    style="width: 400px; height: 600px; border: none; border-radius: 8px;"
    title="Amigo Chat Widget">
</iframe>

4. JavaScript Auto-Resize Helper

For advanced implementations, you can use JavaScript to automatically resize the iframe:

// Auto-resize iframe based on content
function setupAutoResize(iframeId) {
    const iframe = document.getElementById(iframeId);
    
    window.addEventListener('message', function(event) {
        if (event.origin !== 'https://tretuttle.github.io') return;
        
        if (event.data.type === 'resize-iframe') {
            const { width, height } = event.data;
            if (width) iframe.style.width = width + 'px';
            if (height) iframe.style.height = height + 'px';
        }
    });
}

// Usage
setupAutoResize('responsive-widget');
